    """
    A custom tokenizer class that extends Transformers' PreTrainedTokenizer,
    offering entity handling, special token replacement, and advanced batch encoding functionalities.

    Args:
        entity2id (Dict[str, int], optional): A mapping of entity names to IDs. Defaults to None.
        pad_entity_id (int, optional): The ID to use for padding entities. Defaults to None.
        tokenizers (Union[List[PreTrainedTokenizerBase], PreTrainedTokenizerBase]): A list of tokenizers or a single tokenizer. Defaults to None.
        **kwargs: Additional keyword arguments.

    Attributes:
        entity2uid (Dict[str, int]): A mapping of entity names to IDs.
        uid2entity (Dict[int, str]): A mapping of entity IDs to names.
        pad_entity_id (int): The ID used for padding entities.
        tokenizers (List[PreTrainedTokenizerBase]): List of tokenizers.
        entity_pattern (re.Pattern): Regular expression pattern for entities.
    """

[docs]    def __init__(self,
                 entity2id: Dict[str, int]=None,
                 pad_entity_id: int=None,
                 tokenizers = Union[List[PreTrainedTokenizerBase], PreTrainedTokenizerBase],
                 **kwargs):
        super().__init__(
            **kwargs
        )

        self.entity2uid = entity2id
        self.uid2entity = {v: k for k, v in entity2id.items()}
        self.pad_entity_id = pad_entity_id if pad_entity_id is not None else max(self.entity2uid.values()) + 1
        self.tokenizers = tokenizers if isinstance(tokenizers, List) else [tokenizers]
        self.entity_pattern = re.compile(ENTITY_PATTERN)


    def unk_token(self) -> str:
        return "<unk>"

    def unk_token_id(self) -> Optional[int]:
        return -1

    @property
    def vocab_size(self) -> int:
        return len(self.entity2uid)
    
    def _convert_token_to_id(self, token: str) -> int:
        return self.entity2uid[token] if token in self.entity2uid else self.unk_token_id

    def _convert_id_to_token(self, index: int) -> str:
        return self.uid2entity[index] if index in self.uid2entity else self.unk_token

    @staticmethod
    def to_dict(encoding: BatchEncoding, prefix=''):
        return {
            f'{prefix}.input_ids': encoding.data['input_ids'],
            f'{prefix}.attention_mask': encoding.data['attention_mask']
        }
    @staticmethod
    def mergeEncoding(encodings: List[BatchEncoding]) -> BatchEncoding:
        if len(encodings) == 0:
            return BatchEncoding({})
        res = encodings[0].copy()
        for encoding in encodings[1:]:
            res.data.update(encoding.data)
        return res

    pattern_sep_token = re.compile(SEP_TOKEN)
    pattern_cls_token = re.compile(CLS_TOKEN)
    pattern_eos_token = re.compile(EOS_TOKEN)


[docs]    def replace_special_tokens(self, text: str) -> List[str]:
        """
        Replace the cls token, sep token and eos token for each tokenizer

        Args:
            text (str): The input text to be replaced.

        Returns:
            List[str]: A list of text, each processed for one tokenizer.

        Example:
            >>> text = "This is a sample sentence [CLS] with [SEP] special tokens [EOS]."
            >>> replace_special_tokens(text)
            ['This is a sample sentence with special tokens .']
        """
        def replace_for_tokenizer(text: str, tokenizer: PreTrainedTokenizerBase):
            text = self.replace_cls_token(text, tokenizer)
            text = self.replace_sep_token(text, tokenizer)
            text = self.replace_eos_token(text, tokenizer)
            return text

        return [replace_for_tokenizer(text, tokenizer) for tokenizer in self.tokenizers]


    def replace_sep_token(self, text, tokenizer: PreTrainedTokenizerBase):
        token = str(tokenizer._sep_token or tokenizer.eos_token)
        text = self.pattern_sep_token.sub(token, text)
        return text

    def replace_cls_token(self, text, tokenizer: PreTrainedTokenizerBase):
        token = str(tokenizer._cls_token or '')
        text = self.pattern_cls_token.sub(token, text)
        return text
    def replace_eos_token(self, text, tokenizer: PreTrainedTokenizerBase):
        token = str(tokenizer._eos_token or ' ')
        text = self.pattern_eos_token.sub(token, text)
        return text


[docs]    def encodes(self, encode_funcs: List[Callable], texts: List[Union[str,List[str]]], *args, **kwargs) -> List[BatchEncoding]:
        """
        This function is called to apply encoding functions from different tokenizers. It will be
        used by both `encode_plus` and `batch_encode_plus`.

        If you want to call different tokenizers with different arguments, override this method.

        Args:
            encode_funcs (List[Callable]): Encoding functions from `self.tokenizers`.
            texts (List[Union[str, List[str]]]): Processed text for each encoding function.
            **kwargs: Additional keyword arguments.

        Returns:
            List[BatchEncoding]: A list of BatchEncoding objects, the length of the list is the same as the number of tokenizer.
        """
        assert len(encode_funcs) == len(texts)
        return [func(text, *args, **kwargs) for text, func in zip(texts, encode_funcs)]

            str: The processed text.

    def batch_encode_plus(
        self,
        batch_text_or_text_pairs: List[str],
        *args,
        **kwargs
    ) -> BatchEncoding:
        # preprocess
        batch_text = map(self.preprocess, batch_text_or_text_pairs)
        # process entity
        processed_results = [self.process_entities(text) for text in batch_text]
        processed_text, batch_entities = map(list, zip(*processed_results))
        # replace sep token for each tokenzizer
        texts = [list(text) for text in zip(*[self.replace_special_tokens(s) for s in processed_text])]
        # call the encodes function on the list of tokenizer and the list of text
        encodings = self.encodes([tokenizer.batch_encode_plus for tokenizer in self.tokenizers], texts, *args, **kwargs)
        encodings = self.mergeEncoding(encodings)
        # add entities to the encodings
        if kwargs.get('padding') == True:
            batch_entities = padded_tensor(batch_entities, pad_idx=self.pad_entity_id)
            if kwargs.get('return_tensors') is None:
                batch_entities = batch_entities.tolist()
        encodings.data['entities'] = batch_entities
        return encodings

    def encode_plus(
        self,
        text: str,
        *args,
        **kwargs
    ) -> BatchEncoding:
        text = self.preprocess(text)
        processed_text, entities = self.process_entities(text)
        texts = self.replace_special_tokens(processed_text)
        encodings = self.encodes([tokenizer.encode_plus for tokenizer in self.tokenizers], texts, *args, **kwargs)
        encodings = self.mergeEncoding(encodings)
        if kwargs.get('return_tensors') == 'pt':
            entities = torch.tensor(entities, dtype=torch.long)
        encodings.data['entities'] = entities
        return encodings


    def process_entities(self, text: str) -> Tuple[str, List[Entity]]:
        entities = []
        for m in reversed(list(self.entity_pattern.finditer(text))): # use reverse to avoid breaking the span
            start, end = m.span()
            entity_name = m.group(1)
            text = text[:start] + entity_name + text[end:]
            if self.entity2uid is not None and entity_name in self.entity2uid:
                entities.append(self.entity2uid[entity_name])
        entities = entities[::-1]
        return text, entities

        vocab = json.load(open(os.path.join(pretrained_model_name_or_path, 'vocab.json')))
        kwargs.update({'entity2id': vocab})
        return cls(*args, **kwargs)



[docs]def padded_tensor(
        items: List[Union[List[int], torch.Tensor]],
        pad_idx: int = 0,
        pad_tail: bool = True,
) -> torch.Tensor:
    """Create a padded matrix from an uneven list of lists. Matrix is right-padded (filled to the right) by default, but can be
    left padded if the flag is set to True. Matrix can also be placed on cuda automatically.

    Args:
        items (list[iter[int]]): List of items.
        pad_idx (int): The value to use for padding.
        pad_tail (bool): If True, the matrix is right-padded; if False, it is left-padded.
        max_len (int): If None, the max length is the maximum item length.

    Returns:
        Tensor[int64]: Padded tensor.

    Example:
        >>> items = [[1, 2, 3], [4, 5], [6, 7, 8, 9]]
        >>> pad_idx = 0
        >>> create_padded_matrix(items, pad_idx, pad_tail=True, max_len=None)
        tensor([[1, 2, 3, 0],
                [0, 4, 5, 0],
                [6, 7, 8, 9]], dtype=torch.int64)
    """
    # number of items
    n = len(items)
    # length of each item
    lens: List[int] = [len(item) for item in items]
    # max in time dimension
    t = max(lens)
    # if input tensors are empty, we should expand to nulls
    t = max(t, 1)

    output = torch.full((n, t), fill_value=pad_idx, dtype=torch.long)

    for i, (item, length) in enumerate(zip(items, lens)):
        if length == 0:
            continue
        if not isinstance(item, torch.Tensor):
            item = torch.tensor(item, dtype=torch.long)
        if pad_tail:
            output[i, :length] = item
        else:
            output[i, t - length:] = item

    return output
